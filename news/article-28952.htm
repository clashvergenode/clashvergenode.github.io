首先要配置好路由

在app.vue添加下面代码

export default {   provide() {     return {       reload: this.reload,     }   },   data() {     return {       isRouterAlive: true,     }   },   methods: {     reload() {       this.isRouterAlive = false       this.$nextTick(function() {         this.isRouterAlive = true       })     },   }, }

并在template引入

<router-view v-if="isRouterAlive"/>

在需要触发的方法下添加

this.reload()
